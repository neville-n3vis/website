import type { MetadataRoute } from 'next'

export default function robots(): MetadataRoute.Robots {
  return {
    rules: {
      userAgent: '*',
      allow: '/',
      disallow: ['/api/', '/admin/', '/dashboard/'], // Block these from search results
    },
    sitemap: 'https://www.n3vis.com/sitemap.xml',
  }
}
